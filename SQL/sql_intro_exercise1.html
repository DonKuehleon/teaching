require(RPostgreSQL)

drv = dbDriver('PostgreSQL')
con = dbConnect(
  drv = drv,
  host = 'localhost',
  port = 5432,
  dbname = 'test',
  user = 'postgres',
  password = 'whatever'
)

dbWriteTable(con, name = 'sites', value = sit,
             overwrite = TRUE, row.names = FALSE)

dbDisconnect(con)
dbUnloadDriver(drv)
